44async def get_or_create_user(
45 db: Connection,
46 user_data: AuthUser,
47) -> DbUser:
48 """Get user from User table if exists, else create."""
49 try:
50 upsert_sql = """
51 WITH upserted_user AS (
52 INSERT INTO users (
53 sub,
54 username,
55 email_address,
56 profile_img,
57 role,
58 registered_at
59 )
60 VALUES (
61 %(user_sub)s,
62 %(username)s,
63 %(email_address)s,
64 %(profile_img)s,
65 %(role)s,
66 NOW()
67 )
68 ON CONFLICT (sub)
69 DO UPDATE SET
70 profile_img = EXCLUDED.profile_img,
71 last_login_at = NOW()
72 RETURNING sub, username, email_address, profile_img, role
73 )
74
75 SELECT
76 u.sub,
77 u.username,
78 u.email_address,
79 u.profile_img,
80 u.role,
81
82 -- Aggregate the organisation IDs managed by the user
83 array_agg(
84 DISTINCT om.organisation_id
85 ) FILTER (WHERE om.organisation_id IS NOT NULL) AS orgs_managed,
86
87 -- Aggregate project roles for the user, as project:role pairs
88 jsonb_object_agg(
89 ur.project_id,
90 COALESCE(ur.role, 'MAPPER')
91 ) FILTER (WHERE ur.project_id IS NOT NULL) AS project_roles
92
93 FROM upserted_user u
94 LEFT JOIN user_roles ur ON u.sub = ur.user_sub
95 LEFT JOIN organisation_managers om ON u.sub = om.user_sub
96 GROUP BY
97 u.sub,
98 u.username,
99 u.email_address,
100 u.profile_img,
101 u.role;
102 """
103
104 async with db.cursor(row_factory=class_row(DbUser)) as cur:
105 await cur.execute(
106 upsert_sql,
107 {
108 "user_sub": user_data.sub,
109 "username": user_data.username,
110 "email_address": user_data.email,
111 "profile_img": user_data.profile_img or "",
112 "role": UserRole(user_data.role).name,
113 },
114 )
115 db_user_details = await cur.fetchone()
116
117 if not db_user_details:
118 raise HTTPException(
119 status_code=HTTPStatus.NOT_FOUND,
120 detail=f"User ({user_data.sub}) could not be inserted in db",
121 )
122
123 return db_user_details
124
125 except Exception as e:
126 await db.rollback()
127 log.exception(f"Exception occurred: {e}", stack_info=True)
128 raise HTTPException(status_code=HTTPStatus.BAD_REQUEST, detail=str(e)) from e

131async def process_inactive_users(
132 db: Connection,
133):
134 """Identify inactive users, send warnings, and delete accounts."""
135 now = datetime.now(timezone.utc)
136 warning_thresholds = [
137 (now - timedelta(days=INACTIVITY_THRESHOLD - days))
138 for days in WARNING_INTERVALS
139 ]
140 deletion_threshold = now - timedelta(days=INACTIVITY_THRESHOLD)
141
142 async with db.cursor() as cur:
143 # Users eligible for warnings
144 for days, warning_date in zip(
145 WARNING_INTERVALS, warning_thresholds, strict=False
146 ):
147 async with db.cursor(row_factory=class_row(DbUser)) as cur:
148 await cur.execute(
149 """
150 SELECT sub, username, last_login_at
151 FROM users
152 WHERE last_login_at < %(warning_date)s
153 AND last_login_at >= %(next_warning_date)s;
154 """,
155 {
156 "warning_date": warning_date,
157 "next_warning_date": warning_date - timedelta(days=7),
158 },
159 )
160 users_to_warn = await cur.fetchall()
161
162 for user in users_to_warn:
163 if SVC_OSM_TOKEN:
164 await send_warning_email_or_osm(
165 user.sub, user.username, days, SVC_OSM_TOKEN
166 )
167 else:
168 log.warning(
169 f"The SVC_OSM_TOKEN is not set on this server. "
170 f"Cannot send emails to inactive users: "
171 f"{', '.join(user.username for user in users_to_warn)}"
172 )
173
174 # Users eligible for deletion
175 async with db.cursor(row_factory=class_row(DbUser)) as cur:
176 await cur.execute(
177 """
178 SELECT sub, username
179 FROM users
180 WHERE last_login_at < %(deletion_threshold)s;
181 """,
182 {"deletion_threshold": deletion_threshold},
183 )
184 users_to_delete = await cur.fetchall()
185
186 for user in users_to_delete:
187 log.info(f"Deleting user {user.username} due to inactivity.")
188 await DbUser.delete(db, user.sub)

257async def get_paginated_users(
258 db: Connection,
259 page: int,
260 results_per_page: int,
261 search: Optional[str] = None,
262 signin_type: Literal["osm", "google"] = "osm",
263) -> dict:
264 """Helper function to fetch paginated users with optional filters."""
265 # Get subset of users
266 users = await DbUser.all(db, search=search, signin_type=signin_type) or []
267 start_index = (page - 1) * results_per_page
268 end_index = start_index + results_per_page
269
270 if not users:
271 paginated_users = []
272 else:
273 paginated_users = users[start_index:end_index]
274
275 pagination = await get_pagination(
276 page, len(paginated_users), results_per_page, len(users)
277 )
278 return {"results": paginated_users, "pagination": pagination}
