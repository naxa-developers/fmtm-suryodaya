163@router.get("/logout")
164async def logout():
165 """Reset httpOnly cookie to sign out user."""
166 response = Response(status_code=HTTPStatus.OK)
167 # Reset all cookies (logout)
168 fmtm_cookie_name = settings.cookie_name
169 refresh_cookie_name = f"{fmtm_cookie_name}_refresh"
170 osm_cookie_name = f"{fmtm_cookie_name}_osm"
171
172 cookie_names = [
173 fmtm_cookie_name,
174 refresh_cookie_name,
175 osm_cookie_name,
176 ]
177
178 response = await expire_cookies(response, cookie_names)
179 return response

199@router.get("/refresh/management", response_model=FMTMUser)
200async def refresh_management_cookies(
201 request: Request,
202 current_user: Annotated[AuthUser, Depends(login_required)],
203):
204 """Uses the refresh token to generate a new access token.
205
206 This endpoint is specific to the management desktop frontend.
207 Authentication is required.
208 If signed in with login method other than OSM, the user will be logged out and
209 a forbidden status will be returned.
210
211 NOTE this endpoint has no db calls and returns in ~2ms.
212 """
213 # Only allow login via OSM for management frontend
214 # and revoke cookies if service account set via mapper frontend
215 user_sub = current_user.sub.lower()
216 if "osm" not in user_sub or current_user.username == "svcfmtm":
217 response = Response(
218 status_code=HTTPStatus.FORBIDDEN,
219 content="Please log in using OSM for management access.",
220 )
221 cookie_names = [
222 settings.cookie_name,
223 f"{settings.cookie_name}_refresh",
224 ]
225
226 response = await expire_cookies(response, cookie_names)
227 return response
228
229 return await refresh_cookies(
230 request,
231 current_user,
232 settings.cookie_name,
233 f"{settings.cookie_name}_refresh",
234 )

237@router.get("/refresh/mapper", response_model=Optional[FMTMUser])
238async def refresh_mapper_token(
239 request: Request,
240 current_user: Annotated[AuthUser, Depends(public_endpoint)],
241):
242 """Uses the refresh token to generate a new access token.
243
244 This endpoint is specific to the mapper mobile frontend.
245 By default the user will be logged in with a service account.
246 Authentication is optional, if the user wishes to be attributed for contributions.
247
248 NOTE this endpoint has no db calls and returns in ~2ms.
249 """
250 try:
251 # If standard login cookie is passed, use that
252 response = await refresh_cookies(
253 request,
254 current_user,
255 settings.cookie_name,
256 f"{settings.cookie_name}_refresh",
257 )
258 return response
259 except HTTPException:
260 # NOTE we allow for token verification to fail for the main cookie
261 # and fallback to to generate a temp auth cookie
262 pass
263
264 # Refresh the temp cookies (we must re-create the 'sub' field)
265 temp_jwt_details = {
266 **current_user.model_dump(exclude=["id"]),
267 "sub": current_user.sub,
268 "aud": settings.FMTM_DOMAIN,
269 "iat": int(time()),
270 "exp": int(time()) + 86400, # set token expiry to 1 day
271 }
272
273 fmtm_token, refresh_token = create_jwt_tokens(temp_jwt_details)
274 # NOTE be sure to not append content=current_user.model_dump() to this JSONResponse
275 # as we want the login state on the frontend to remain empty (allowing the user to
276 # log in via OSM instead / override)
277 response = JSONResponse(status_code=HTTPStatus.OK, content=temp_jwt_details)
278 return set_cookies(
279 response,
280 fmtm_token,
281 refresh_token,
282 )
