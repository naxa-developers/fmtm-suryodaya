44async def get_project_task_activity(
45 db: Connection,
46 project_id: int,
47 days: int,
48) -> task_schemas.TaskEventCount:
49 """Get number of tasks mapped and validated for project.
50
51 Args:
52 project_id (int): The ID of the project.
53 days (int): The number of days to consider for the
54 task activity (default: 10).
55 db (Connection): The database connection.
56
57 Returns:
58 list[task_schemas.TaskEventCount]: A list of task event counts.
59 """
60 end_date = timestamp() - timedelta(days=days)
61
62 sql = """
63 SELECT
64 to_char(created_at::date, 'dd/mm/yyyy') as date,
65 COUNT(*) FILTER (WHERE state = 'UNLOCKED_DONE') AS validated,
66 COUNT(*) FILTER (WHERE state = 'UNLOCKED_TO_VALIDATE') AS mapped
67 FROM
68 task_events
69 WHERE
70 project_id = %(project_id)s
71 AND created_at >= %(end_date)s
72 GROUP BY
73 created_at::date
74 ORDER BY
75 created_at::date;
76 """
77
78 async with db.cursor(row_factory=class_row(task_schemas.TaskEventCount)) as cur:
79 await cur.execute(sql, {"project_id": project_id, "end_date": end_date})
80 return await cur.fetchall()

83async def trigger_unlock_tasks(db: Connection):
84 """Function to unlock_old_locked_tasks manually."""
85 active_projects_query = """
86 SELECT DISTINCT project_id
87 FROM task_events
88 WHERE created_at >= NOW() - INTERVAL '7 days'
89 """
90 async with db.cursor() as cur:
91 await cur.execute(active_projects_query)
92 active_projects = await cur.fetchall()
93
94 time_now = datetime.now(timezone.utc)
95 threedaysago = (time_now - timedelta(days=3)).strftime("%Y-%m-%dT%H:%M:%S.%fZ")
96 onehourago = (time_now - timedelta(hours=1)).strftime("%Y-%m-%dT%H:%M:%S.%fZ")
97
98 for (project_id,) in active_projects:
99 project = await DbProject.one(db, project_id, True)
100
101 recent_project_entities = await get_entities_data(
102 project.odk_credentials, project.odkid, filter_date=threedaysago
103 )
104 # If there are recent entity updates, skip this project
105 if recent_project_entities:
106 continue
107
108 await reset_entities_status(db, project.id, onehourago)
109
110 # Only unlock tasks if there are no recent entity updates
111 await unlock_old_locked_tasks(db, project.id)

114async def reset_entities_status(db, project_id, filter_date):
115 """Reset status for entities that have been 'open in ODK' for more than 1hr."""
116 project = await DbProject.one(db, project_id, True)
117 recent_opened_entities = await get_entities_data(
118 project.odk_credentials,
119 project.odkid,
120 filter_date=filter_date,
121 )
122 for entity in recent_opened_entities:
123 if entity["status"] != str(EntityState.OPENED_IN_ODK):
124 continue
125 await update_entity_mapping_status(
126 project.odk_credentials,
127 project.odkid,
128 entity["id"],
129 f"Task {entity['task_id']} Feature {entity['osm_id']}",
130 str(EntityState.READY),
131 )
132
133 # Sync ODK entities in our database
134 project_entities = await get_entities_data(project.odk_credentials, project.odkid)
135 await DbOdkEntities.upsert(db, project.id, project_entities)

138async def unlock_old_locked_tasks(db, project_id):
139 """Unlock tasks locked for more than 3 days."""
140 unlock_query = """
141 WITH svc_user AS (
142 SELECT sub, username
143 FROM users
144 WHERE username = 'svcfmtm'
145 ),
146 recent_events AS (
147 SELECT DISTINCT ON (t.id, t.project_id)
148 t.id AS task_id,
149 t.project_id,
150 the.created_at AS last_event_time,
151 the.event AS last_event
152 FROM tasks t
153 JOIN task_events the
154 ON t.id = the.task_id
155 AND t.project_id = the.project_id
156 AND the.comment IS NULL
157 WHERE t.project_id = %(project_id)s
158 ORDER BY t.id, t.project_id, the.created_at DESC
159 ),
160 filtered_events AS (
161 SELECT *
162 FROM recent_events
163 WHERE (
164 (last_event = 'ASSIGN' AND last_event_time < NOW() - INTERVAL '3 days')
165 OR
166 (last_event = 'MAP' AND last_event_time < NOW() - INTERVAL '3 hours')
167 )
168 )
169 INSERT INTO task_events (
170 event_id,
171 task_id,
172 project_id,
173 event,
174 user_sub,
175 state,
176 created_at,
177 username
178 )
179 SELECT
180 gen_random_uuid(),
181 fe.task_id,
182 fe.project_id,
183 'RESET'::taskevent,
184 svc.sub,
185 'UNLOCKED_TO_MAP'::mappingstate,
186 NOW(),
187 svc.username
188 FROM filtered_events fe
189 CROSS JOIN svc_user svc;
190 """
191 async with db.cursor() as cur:
192 await cur.execute(unlock_query, {"project_id": project_id})

195async def send_task_assignment_notifications(
196 request: Request,
197 osm_auth: Auth,
198 team: DbProjectTeam,
199 project_user: ProjectUserDict,
200 task: DbTask,
201 assignee_sub: str,
202):
203 """Send OSM notifications to team members or assignee for task assignment."""
204 osm_token = get_osm_token(request, osm_auth)
205 project = project_user.get("project")
206 user = project_user.get("user")
207
208 project_url = f"{settings.FMTM_DOMAIN}/project/{project.id}"
209 if not project_url.startswith("http"):
210 project_url = f"https://{project_url}"
211
212 if assignee_sub:
213 assignee_message_content = dedent(f"""
214 You have been assigned to task **{task.project_task_index}** in project
215 **{project.name}** by **{user.username}**.
216
217 [View Project]({project_url})
218
219 Thank you for being a part of our platform!
220 """)
221
222 send_osm_message(
223 osm_token=osm_token,
224 title=f"Task Assignment in {project.name}",
225 body=assignee_message_content,
226 osm_sub=assignee_sub,
227 )
228
229 log.info(f"OSM notification sent to {assignee_sub} for task {task.id}")
230
231 elif team:
232 team_message_content = dedent(f"""
233 You have been assigned to task **{task.project_task_index}** in project
234 **{project.name}** by **{user.username}** as a part of team
235 **{team.team_name}**.
236
237 [View Project]({project_url})
238
239 Thank you for being a part of our platform!
240 """)
241
242 for user in team.users:
243 send_osm_message(
244 osm_token=osm_token,
245 title=f"Task Assignment in {project.name}",
246 body=team_message_content,
247 osm_sub=user["sub"],
248 )
249
250 log.info(f"OSM notifications sent to {team.users} for task {task.id}")
