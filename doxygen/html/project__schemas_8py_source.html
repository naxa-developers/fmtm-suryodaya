116 @field_validator("hashtags", mode="before")
117 @classmethod
118 def validate_hashtags(
119 cls,
120 hashtags: Optional[str | list[str]],
121 ) -> Optional[list[str]]:
122 """Validate hashtags.
123
124 - Receives a string and parsed as a list of tags.
125 - Commas or semicolons are replaced with spaces before splitting.
126 - Add '#' to hashtag if missing.
127 """
128 if hashtags is None:
129 return None
130
131 if isinstance(hashtags, str):
132 hashtags = hashtags.replace(",", " ").replace(";", " ")
133 hashtags_list = hashtags.split()
134 else:
135 hashtags_list = hashtags
136
137 # Add '#' to hashtag strings if missing
138 return [
139 f"#{hashtag}" if hashtag and not hashtag.startswith("#") else hashtag
140 for hashtag in hashtags_list
141 ]

151 @field_validator("outline", mode="before")

156 ) -> Optional[Polygon]:
157 """Parse any format geojson into a single Polygon.
158
159 NOTE we run this in mode='before' to allow parsing as Feature first.
160 """
161 if value is None:
162 return None

164 # geojson_pydantic.GeometryCollection
165 # FIXME update this to remove the Featcol parsing at some point
166 featcol = geojson_to_featcol(value)
167 merged_geojson = merge_polygons(featcol, True)
168 return merged_geojson.get("features")[0].get("geometry")

319 @computed_field
320 @property
321 def format(self) -> Optional[str]:
322 """Get the basemap format from file extension."""
323 if not self.urlurl:
324 return None
325 return Path(self.urlurl).suffix[1:]
326
327 @computed_field
328 @property
329 def mimetype(self) -> Optional[str]:
330 """Set the mimetype based on the extension."""
331 if not self.urlurl:
332 return None
333
334 file_format = Path(self.urlurl).suffix[1:]
335 if file_format == "mbtiles":
336 return "application/vnd.mapbox-vector-tile"
337 elif file_format == "pmtiles":
338 return "application/vnd.pmtiles"
339 else:
340 return "application/vnd.sqlite3"
